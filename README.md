## Install

```bash
npm install --save lumina-design
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'lumina-design'
import 'lumina-design/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
